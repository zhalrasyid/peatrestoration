public function get_wdpa($wdpapoly_id)
		{
			$this->db->select('wdpapoly_id, st_asgeojson(st_transform(the_geom,4326)) as the_geom');
			$this->db->from('wdpa_poly');
			$this->db->where('wdpapoly_id', $wdpapoly_id);
			$query = $this->db->get();
			return $query->row_array();					
		}
